from rest_framework import serializers
from .models import User


class RegisterSerializer(serializers.ModelSerializer):

    class Meta:

        model = User

        fields = (
            'id',
            'fullname',
            'username',
            'password',
            'gender',
            'role'
        )

        extra_kwargs = {
            'password': {'write_only': True}
        }

    def validate_role(self, value):

        if value in ['admin', 'tutor']:

            raise serializers.ValidationError(
                "Cannot assign admin/tutor role"
            )

        return value

    def create(self, validated_data):

        return User.objects.create_user(
            username=validated_data['username'],
            password=validated_data['password'],
            fullname=validated_data['fullname'],
            gender=validated_data['gender'],
            role='learner'
        )
Why?

Question says:

Learner should not assign himself Admin or Tutor role

This validation solves it.

from .models import Enrolment
class EnrolmentSerializer(
    serializers.ModelSerializer
):

    class Meta:

        model = Enrolment

        fields = '__all__'

    def validate(self, data):

        course = data['course']

        count = Enrolment.objects.filter(
            course=course,
            is_deleted=False
        ).count()

        if count >= course.capacity:

            raise serializers.ValidationError(
                "Course full"
            )

        return data
Why?

Question says:

Capacity reached -> no enrolment

This is where you enforce it.

def get_queryset(self):

import { useState } from "react";
import api from "../api/axios";

function LoginPage() {

    const [username,setUsername] =
        useState("");

    const [password,setPassword] =
        useState("");

    const login = async () => {

        try{

            const response =
                await api.post(
                    "auth/login/",
                    {
                        username,
                        password
                    }
                );

            localStorage.setItem(
                "access",
                response.data.access
            );

            alert("Login success");

        }catch(error){

            alert("Invalid login");
        }
    };

    return (

        <div>

            <h2>Login</h2>

            <input
                placeholder="Username"
                onChange={
                    e =>
                    setUsername(
                        e.target.value
                    )
                }
            />

            <input
                type="password"
                placeholder="Password"
                onChange={
                    e =>
                    setPassword(
                        e.target.value
                    )
                }
            />

            <button onClick={login}>
                Login
            </button>

        </div>
    );
}

export default LoginPage;

import { useEffect } from "react";
import { useState } from "react";

import api from "../api/axios";

function CoursePage(){

    const [courses,setCourses] =
        useState([]);

    useEffect(()=>{

        const loadCourses =
            async ()=>{

            const token =
                localStorage.getItem(
                    "access"
                );

            const response =
                await api.get(
                    "courses/",
                    {
                        headers:{
                            Authorization:
                            `Bearer ${token}`
                        }
                    }
                );

            setCourses(
                response.data
            );
        };

        loadCourses();

    },[]);

    return(

        <div>

            <h2>Courses</h2>

            {
                courses.map(course=>(

                    <div
                        key={course.id}
                    >

                        <h3>
                            {course.title}
                        </h3>

                        <p>
                            Capacity:
                            {course.capacity}
                        </p>

                    </div>
                ))
            }

        </div>
    );
}

export default CoursePage;

STEP 8: REGISTER USER
service/UserService.java
@Service
@RequiredArgsConstructor
public class UserService {

    private final UserRepository repo;

    public User register(
            RegisterRequest request){

        if(
            "admin".equalsIgnoreCase(
                request.getRole()
            ) ||

            "tutor".equalsIgnoreCase(
                request.getRole()
            )
        ){

            throw new RuntimeException(
                "Role not allowed"
            );
        }

        User user = new User();

        user.setFullname(
            request.getFullname()
        );

        user.setUsername(
            request.getUsername()
        );

        user.setPassword(
            request.getPassword()
        );

        user.setGender(
            request.getGender()
        );

        user.setRole("learner");

        return repo.save(user);
    }
}
Business Rule Covered
Learner cannot assign Admin/Tutor role

STEP 12: ENROLMENT SERVICE
service/EnrolmentService.java
@Service
@RequiredArgsConstructor
public class EnrolmentService {

    private final EnrolmentRepository repo;

    private final CourseRepository courseRepo;

    public Enrolment create(
            Long courseId,
            User learner){

        Course course =
            courseRepo.findById(courseId)
            .orElseThrow();

        long count =
            repo.countByCourseAndDeletedFalse(
                course
            );

        if(
            count >=
            course.getCapacity()
        ){

            throw new RuntimeException(
                "Course full"
            );
        }

        Enrolment enrolment =
            new Enrolment();

        enrolment.setLearner(
            learner
        );

        enrolment.setCourse(
            course
        );

        return repo.save(
            enrolment
        );
    }
}
Business Rule Covered
Course capacity reached

import random


def calculate_score(job):

    resume_score = random.randint(
        1,
        10
    )

    if job.location.lower() == 'remote':

        location_points = 3

    elif job.location.upper() == 'DAR ES SALAAM':

        location_points = 2

    else:

        location_points = 1

    final_score = (
        resume_score +
        location_points
    )

    return (
        resume_score,
        location_points,
        final_score
    )

import { useState } from "react";
import api from "../api";

function Login() {

    const [username,setUsername] =
        useState("");

    const [password,setPassword] =
        useState("");

    const login = async () => {

        const res =
            await api.post(
                "auth/login/",
                {
                    username,
                    password
                }
            );

        localStorage.setItem(
            "token",
            res.data.access
        );
    };

    return (
        <>
            <input
                onChange={
                    e =>
                    setUsername(
                        e.target.value
                    )
                }
            />

            <input
                type="password"
                onChange={
                    e =>
                    setPassword(
                        e.target.value
                    )
                }
            />

            <button onClick={login}>
                Login
            </button>
        </>
    );
}

export default Login;
