class User(AbstractUser):

        ('learner', 'Learner'),

    GENDER_CHOICES = (
        ('male', 'Male'),

        max_length=20,

        max_length=20,
        choices=STATUS_CHOICES
    )

    created_at = models.DateTimeField(
        auto_now_add=True

class Enrolment(models.Model):

    learner = models.ForeignKey(
        User,
        on_delete=models.CASCADE
    )

    course = models.ForeignKey(
        Course,
        on_delete=models.CASCADE
    )

    created_at = models.DateTimeField(
        auto_now_add=True
    )

    is_deleted = models.BooleanField(
        default=False
    )

    class Meta:

        unique_together = (
            'learner',
            'course'
        )
Why add is_deleted?

Interview question:

justify hard delete or soft delete

Use soft delete.

is_deleted=True

instead of deleting.

Answer:

Academic systems require audit history.

STEP 8: USER REGISTRATION SERIALIZER
File
accounts/serializers.py
from rest_framework import serializers
from .models import User


class RegisterSerializer(serializers.ModelSerializer):

    class Meta:

        model = User

        fields = (
            'id',
            'fullname',
            'username',
            'password',
            'gender',
            'role'
        )

        extra_kwargs = {
            'password': {'write_only': True}
        }

    def validate_role(self, value):

        if value in ['admin', 'tutor']:

            raise serializers.ValidationError(
                "Cannot assign admin/tutor role"
            )

        return value

    def create(self, validated_data):

        return User.objects.create_user(
            username=validated_data['username'],
            password=validated_data['password'],
            fullname=validated_data['fullname'],
            gender=validated_data['gender'],
            role='learner'
        )
Why?

Question says:

Learner should not assign himself Admin or Tutor role

This validation solves it.

STEP 14: ENROLMENT SERIALIZER
File
courses/serializers.py

Add below CourseSerializer.

from .models import Enrolment
class EnrolmentSerializer(
    serializers.ModelSerializer
):

    class Meta:

        model = Enrolment

        fields = '__all__'

    def validate(self, data):

        course = data['course']

        count = Enrolment.objects.filter(
            course=course,
            is_deleted=False
        ).count()

        if count >= course.capacity:

            raise serializers.ValidationError(
                "Course full"
            )

        return data
Why?

Question says:

Capacity reached -> no enrolment

This is where you enforce it.

STEP 15: ENROLMENT CRUD
File
courses/views.py
class EnrolmentViewSet(
    ModelViewSet
):

    serializer_class = EnrolmentSerializer
Role-based listing
def get_queryset(self):

    user = self.request.user

    if user.role == 'admin':

        return Enrolment.objects.filter(
            is_deleted=False
        )

    elif user.role == 'tutor':

        return Enrolment.objects.filter(
            course__tutor=user,
            is_deleted=False
        )

    return Enrolment.objects.filter(
        learner=user,
        is_deleted=False
    )
Why?

Question says:

Everyone sees enrolments according to role

Admin:

all enrolments

Tutor:

enrolments in his courses

Learner:

his enrolments only
STEP 16: SOFT DELETE

Inside EnrolmentViewSet

def perform_destroy(
    self,
    instance
):

    instance.is_deleted = True

    instance.save()
Interview explanation

Instead of:

instance.delete()

We do:

instance.is_deleted = True

Because:

Academic records must remain available for auditing.
